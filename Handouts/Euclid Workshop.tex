\begin{custom-big}[2011 Euclid Q3b, Functional Equation]
The functions $f$ and $g$ satisfy
\begin{align*}
f(x) + g(x) &= 3x + 5\\
f(x) - g(x) &= 5x + 7
\end{align*}
for all values of $x$. Determine the value of $2f(2)g(2)$.
\end{custom-big}

\begin{solution}
Add up the two equations
\begin{align*}
    2f(x)&=8x+12\\
    f(x)&=4x+6\\
    g(x)&=-x-1\\
    f(2)&=14\\
    g(2)&=-3\\
    \therefore 2f(2)g(2)&=-84
\end{align*}
\end{solution}

\begin{custom-big}[2011 Euclid Q7b, Logarithm]
Determine all values of $x$ for which $2^{\log_{10}\left(x^2\right)}=3\left(2^{1+\log_{10}x}\right)+16$.
\end{custom-big}

\begin{solution}
    \begin{align*}
        2^{2\log_{10}x}&=3\left(2\cdot 2^{\log_{10}x}\right)+16\\
    \end{align*}
    Let $u=2^{\log_{10}x}$, the equation becomes 
    \begin{align*}
        u^2&=6u+16\\
        u&=8,-2\\
        \log_{10}x&=3\\
        x&=1000
    \end{align*}
\end{solution}
