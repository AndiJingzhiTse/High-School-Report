\begin{custom-big}[Problem A.1]
Determine $A, B, C$ such that all of the following functions intersect the point $(2, 2)$:
\begin{align*}
    f_1(x)&=Ax+1 & f_2(x)&=Bx^2+2 & f_3(x)&=Cx^3+3
\end{align*}
\end{custom-big}

\begin{solution}[System of Equations]
\[
\begin{cases}
2A+1=2\\
4B+2=2\\
8C+3=2
\end{cases}\implies
\begin{cases}
A=\frac 12\\
B=0\\
C=-\frac 18
\end{cases}
\]
\end{solution}

\begin{custom-big}[Problem A.2]
Find all $x\in\mathbb R$ that are solutions to this equation: $0=\left(1-x-x^2-\cdots\right)\cdot\left(2-x-x^2-\cdots\right)$
\end{custom-big}

\begin{solution}[Infinite Geometric Sum]\\
The two cases are 
\begin{align*}
    x+x^2+\cdots&=1 & x+x^2+\cdots&=2\\
    \frac{x}{1-x}&=1 & \frac{x}{1-x}&=2\\
    x&=\frac 12&x&=\frac{3}{2}
\end{align*}
\end{solution}

\begin{custom-big}[Problem A.3]
Find the derivative $f'(x)$ of the following function with respect to $x$:
$$\sin \left(\pi ^{\sin x}+\pi ^{\cos x}\right)$$
\end{custom-big}

\begin{solution}[Chain Rule]
\begin{align*}
    f'(x)&=\frac{d}{dx}\left(\pi^{\sin x}+\pi^{\cos x}\right)\cos\left(\pi^{\sin x}+\pi^{\cos x}\right)\\
    &=\left(\cos x\left(\ln \pi\left(\pi^{\sin x}\right)\right)-\sin x\left(\ln\pi\left(\pi^{\cos x}\right)\right)\right)\cos\left(\pi^{\sin x}+\pi^{\cos x}\right)\\
    &=\ln\pi\cos\left(\pi^{\sin x}+\pi^{\cos x}\right)\left(\pi^{\sin x}\cos x-\pi^{\cos x}\sin x\right)
\end{align*}
\end{solution}

\begin{custom-big}[Problem B.1]
Let $H_n$ define the sum of reciprocals of all integers from 1 to $n$:
$$H_n=1+\frac 12+\frac 13+\frac 14+\cdots+\frac 1n$$
Prove the following identity:
$$H_{2n}-H_n=1-\frac 12+\frac 13-\frac 14\pm\cdots+\frac{1}{2n-1}-\frac{1}{2n}$$
\end{custom-big}

\begin{solution}
\begin{align*}
LHS &= H_{2n}-\left(1+\frac 13+\cdots+\frac{1}{2n-1}\right)-\left(H_n-\left(\frac 12+\frac 14+\cdots+ \frac{1}{2n}\right)\right)\\
      &\qquad + \left(1+\frac 13+\cdots+\frac{1}{2n-1}\right)-\left(\frac 12+\frac 14+\cdots+\frac{1}{2n}\right)\\
  &=\frac 12 H_n-\frac 12H_n+\left(1-\frac 12+\frac 13-\frac 14\pm\cdots+\frac{1}{2n-1}-\frac{1}{2n}\right)\\
  &=1-\frac 12+\frac 13-\frac 14\pm\cdots+\frac{1}{2n-1}-\frac{1}{2n}\\
  &=RHS
\end{align*}
\end{solution}

\begin{custom-big}[Problem B.2]
It is well known that squared brackets do not simply square the individual terms:
\begin{align*}
    (1+2)^2&\neq 1^2+2^2\\
    (1+2+3)^2&\neq 1^2+2^2+3^2
\end{align*}
Instead, we add a correction term $\psi$ to make the equations hold true:

\begin{align*}
    (1+2)^2&=1^2+2^2+\psi_2\\
    (1+2+3)^2&=1^2+2^2+3^2+\psi_3\\
    &\ \ \vdots\\
    (1+2+3+\cdots+n)^2&=1^2+2^2+3^2+\cdots+n^2+\psi_n
\end{align*}
Show that the correction term $\psi_n$ has the following form and determine the values of $\alpha$ and $\beta$:
\[
\psi_n=\frac{n^4-n^2}{\alpha}+\frac{n^3-n}{\beta}
\]
\end{custom-big}

\begin{solution}[Series]
\begin{align*}
\left(\frac{n(n+1)}{2}\right)^2&=\frac{n(n+1)(2n+1)}{6}+\psi_n\\
\psi_n&=n(n+1)\left(\frac{2n+1}{6}\right)-n(n+1)\left(\frac{n(n+1)}{4}\right)\\
&=\left(n^2+n\right)\left(\frac{-3n^2+n+2}{12}\right)\\
&=\frac{n^4-n^2}{-4}+\frac{n^3-n}{-6}\\
\therefore (\alpha,\beta)&=(-4,-6)
\end{align*}
\end{solution}


\begin{custom-big}[Problem B.3]
You are given two overlaying squares with side length a. One of the squares is fixed at the
bottom right corner and rotated by an angle of $\alpha$ (see drawing). Find an expression for the enclosed area $A(\alpha)$ between the two squares with respect to the rotation angle $\alpha$.
\begin{center}
    \includegraphics[width=6cm]{Diagrams/Problem B.3.png}
\end{center}
\end{custom-big}

\begin{solution}[Trigonometry]\\
Draw a line passing through the two non $90^\circ$ vertexes of the shaded area. The shaded area equals to twice the area of the triangles formed. 
\begin{align*}
    A(\alpha)&=\ a\left(a\tan\left(45^\circ-\frac{\alpha}{2}\right)\right)\\
    &=\ a^2\tan\left(\frac{1}{2}\left(90^\circ-\alpha\right)\right)\\
    &=\ a^2\left(\frac{1-\sin\alpha}{\cos\alpha}\right)
\end{align*}
\end{solution}

\begin{custom-big}[Problem C.3 (c)]
\[\gamma=\lim_{n\to\infty}\left(H_n-\log(n)\right)=0.5772\dots\]
Use this to prove the following identity:
\[\int_1^\infty\frac{\{x\}}{x^2}dx=1-\gamma\]
\end{custom-big}

\begin{solution}[Integrate by steps, Harmonic Sequence]
\begin{align*}
    \int_1^\infty\frac{\{x\}}{x^2}dx&=\lim_{n\to\infty}\sum_{k=1}^n\int_{k}^{k+1}\frac{x-k}{x^2}dx\\
    &=\lim_{n\to\infty}\sum_{k=1}^n\left(\ln(k+1)-\ln k-\frac{1}{k+1}\right)\\
    &=\lim_{n\to\infty}\left(\ln n-\sum_{k=2}^n\frac 1k\right)\\
    &=\lim_{n\to\infty}\left(\ln n-H_n+1\right)\\
    &=1-\gamma
\end{align*}
QED.
\end{solution}

\begin{custom-big}[Problem C.2]
the proof of Theorem 1, explain why 
\[
\sum_{p\leq x,p\in\mathcal S_{a,b}}\frac 1p=\sum^x_{t=1}\frac{\pi_{a,b}(t)-\pi_{a,b}(t-1)}{t}
\]
\end{custom-big}

\begin{solution}[Bijection]\\
When $t$ is a prime and $at+b$ is also a prime, $\pi_{a,b}(t)-\pi_{a,b}(t-1)=1$, which equals to the numerator on the LHS.\\
When $t$ is not a prime or $at+b$ is not a prime, $\pi_{a,b}(t)-\pi_{a,b}(t-1)=0$. The whole fraction becomes 0, equivalent of not added into the sum. \\
The boundary of LHS and RHS are both $2\leq p\leq x$. This creates a bijection relationships between LHS and RHS and therefore equivalent. 
\end{solution}
