\begin{custom-big}[Problem A.1]
You are given the following three functions:
$$
f(x)=3+3 x \quad g(x)=2+2 x \quad h(x)=5-x
$$

Find $a, b \in \mathbb{R}$ such that $h(x)<f(x)$ and $h(x)>g(x)$ for all $x$ with $a<x<b$.
\end{custom-big}

\begin{solution}
\begin{align*}
    5-x&<3+3x&5-x&>2+2x\\
    4x&>2&3x&<3\\
    x&>\frac 12&x&<1
\end{align*}
\begin{align*}
    \frac 12<x&<1\\
    \therefore (a,b)&=\left(\frac 12,1\right)
\end{align*}
\end{solution}

\begin{custom-big}[Problem A.2]
Find the derivative $f^{\prime}(x)$ of the following function with respect to $x$:
$$
f(x)=\left(1+x^2\right)^x
$$
\end{custom-big}

\begin{solution}
\begin{align*}
f(x) & =e^{x \ln \left(1+x^2\right)} \\
f^{\prime}(x) & =\left(\ln \left(1+x^2\right)+\left(\frac{2 x}{1+x^2}\right)(x)\right)\left(1+x^2\right)^x \\
& =\left(\ln \left(1+x^2\right)+\frac{2 x^2}{1+x^2}\right)\left(1+x^2\right)^x
\end{align*}
\end{solution}

\begin{custom-big}[Problem B.1]
Show that this infinite sum converges and determine its value:
$$
\sum_{n=1}^{\infty} \frac{5^n+5^{n+1}}{5^{2 n+1}}
$$
\end{custom-big}

\begin{solution}
\begin{align*}
&\  \sum_{n=1}^{\infty} \frac{\frac{1}{5}+1}{5^n} \\
=\  & \sum_{n=1}^{\infty}\left(\frac{1}{5}\right)^{n+1}+\sum_{n=1}^{\infty}\left(\frac{1}{5}\right)^n \\
=\  & \frac{6}{5} \sum_{n=1}^{\infty}\left(\frac{1}{5}\right)^n \\
=\  & \frac{6}{5}\left(\frac{\frac{1}{5}}{1-\frac{1}{5}}\right) \\
=\  & \frac{6}{5}\left(\frac{1}{4}\right) \\
=\  & \frac{3}{10}
\end{align*}
\end{solution}

\begin{custom-big}[Problem B.2]
A vertical line $\overline{A B}$ with length $a$ is intersected by a horizontal line at $\frac{2}{3} a$. Another line $\overline{B C}$ with length $a$ is rotated by an angle of $\alpha$ and attached to point $B$. Find an equation for the enclosed area $S(\alpha)$ between the horizontal line and the line $\overline{A C}$ for $\arccos (2 / 3) \leq \alpha \leq \pi$.
\begin{center}
    \includegraphics[width=10cm]{Diagrams/IYMC Pre-Final B2 2023.png}
\end{center}
\end{custom-big}

\begin{solution}
Put the diagram on a Cartesian plane, with $B$ on the origin.

$$
\begin{aligned}
\operatorname{coor} C: & \left(a \cos \left(90^{\circ}-\alpha\right), a \sin \left(90^{\circ}-\alpha\right)\right) \\
= \ & (a \sin \alpha, a \cos \alpha)
\end{aligned}
$$
$l_{A C} \text{ has the equation}:$
$$
\begin{aligned}
y & =\left(\frac{a-a \cos \alpha}{-a \sin \alpha}\right)(x)+a \\
\frac{2 a}{3} & =\left(\frac{\cos \alpha-1}{\sin \alpha}\right) x+a \\
\frac{a}{3} & =\left(\frac{1-\cos \alpha}{\sin \alpha}\right)(x) \\
x & =\frac{a \sin \alpha}{3-3 \cos \alpha} \\
S & =\left(\frac{a \sin \alpha}{3-3 \cos \alpha}\right)\left(\frac{a}{3}\right)\left(\frac{1}{2}\right) \\
S & =\frac{a^2}{18}\left(\frac{\sin \alpha}{1-\cos \alpha}\right) \\
S & =\frac{a^2}{18}\left(\frac{1+\cos \alpha}{\sin \alpha}\right)
\end{aligned}
$$
Alternatively, this can also be written as $$S=\frac{a^2}{18}\left(\csc \alpha+\cot\alpha\right)$$
\end{solution}
