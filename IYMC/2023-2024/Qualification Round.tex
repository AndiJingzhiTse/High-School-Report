\begin{custom-big}[Problem A]
$\text { What are the roots of the function } f(x)=\left(\pi^x-\frac{1}{\pi}\right) \cdot x^\pi \cdot\left(\frac{1}{\pi^2}-\pi^x\right) \text { with } x \in \mathbb{R} \text { ? }$
\end{custom-big}

\begin{solution}
    \begin{align*}
        \pi^x-\frac{1}{\pi}&=0 & x^\pi&=0 &\frac{1}{\pi^2}-\pi^x&=0\\
        \pi^x&=\frac{1}{\pi} & x&=0 & \pi^x&=\frac{1}{\pi^2}\\
        x&=-1 & &&x&=-2
    \end{align*}
$\therefore x=-2,-1,0$
\end{solution}

\begin{custom-big}[Problem B]
$\text { Show that } 2^n-(-1)^n \text { is divisible by } 3 \text { for all positive integers } n \text {. }$
\end{custom-big}

\begin{solution}
    We begin by analyzing the parity of $n$\\
    \underline{If $n\equiv 0\mod 2$:}\\
    \[2^n-(-1)^n=2^n-1=4^\frac{n}{2}-1\]
    Where $\frac n2$ is an integer
    \begin{align*}
        4^\frac{n}{2}-1&\equiv (3+1)^\frac{n}{2}-1 \pmod{3}\\
        &\equiv 1^\frac n2-1\pmod{3}\\
        &\equiv 0\pmod{3}
    \end{align*}
    If $n$ is even, the expression is divisible by 3.
    \bigskip
    
    \underline{If $n\equiv 1\mod 2$:}\\
    \[2^n-(-1)^n=2^n+1=2\left(4^\frac{n-1}{2}\right)+1\]
    Where $\frac{n-1}{2}$ is an integer
    \begin{align*}
       2\left(4^\frac{n-1}{2}\right)+1&\equiv 2\left((3+1)^\frac{n-1}{2}\right)+1 \pmod{3}\\
        &\equiv 2\left(1^\frac n2\right)+1\pmod{3}\\
        &\equiv 3\pmod{3}\\
        &\equiv 0\pmod{3}
    \end{align*}
    If $n$ is odd, the expression is divisible by 3. Combining the two statements proves that the expression is divisble by 3 for all positive integers $n$.
\end{solution}

\begin{custom-big}[Problem C]
Determine the numerical value of the following expression without the use of a calculator:
$$
\log _3\left(\frac{\sin (2 \pi)}{\tan (\pi / 3)}+\sum_{n=0}^{10}\left(\frac{1+\sqrt{1+\sqrt{1}}}{1+\left(1^{-1}+1^{1-1}\right) \cdot \cos (\pi / 4)}-\frac{2^4-2^3}{(1+\sqrt{1})^2}\right)^n\right)
$$
\end{custom-big}

\begin{solution}
    \begin{align*}
    &\log _3\left(\frac{\sin (2 \pi)}{\tan (\pi / 3)}+\sum_{n=0}^{10}\left(\frac{1+\sqrt{1+\sqrt{1}}}{1+\left(1^{-1}+1^{1-1}\right) \cdot \cos (\pi / 4)}-\frac{2^4-2^3}{(1+\sqrt{1})^2}\right)^n\right)\\
        =\ &\log _3\left(\frac{0}{\sqrt 3}+\sum_{n=0}^{10}\left(\frac{1+\sqrt{2}}{1+\left(2\right) \cdot \frac{\sqrt 2}{2}}-\frac{8}{4}\right)^n\right)\\
        =\ &\log _3\left(\sum_{n=0}^{10}\left(-1\right)^n\right)\\
        =\ &\log_31\\
        =\ &0
    \end{align*}
\end{solution}

\begin{custom-big}[Problem D]
Let $\sigma(n)$ be the sum of all positive divisors of the integer $n$ and let $p$ be any prime number. Prove that $\sigma\left(p^m\right)=1+p \cdot \sigma\left(p^{m-1}\right)$ for all positive integers $m$.
\end{custom-big}

\begin{solution}
    First of all, the divisors of $p^m$ are $1,p,p^2,\dots p^m$
    \begin{align*}
    \sigma\left(p^m\right)&=\sum_{k=0}^mp^k=\frac{p^{m+1}-1}{p-1}\\
    \sigma\left(p^{m-1}\right)&=\sum_{k=0}^{m-1}p^k=\frac{p^{m}-1}{p-1}\\
1+p \cdot \sigma\left(p^{m-1}\right)&=\frac{p-1}{p-1}+p\left(\frac{p^{m}-1}{p-1}\right)\\
&=\frac{p^{m+1}-1}{p-1}\\
&= \sigma\left(p^m\right)
    \end{align*}
\end{solution}
