\section{Pressure}
\begin{formula}
Pressure is calculated by the formula:
\[P=\frac FA\]
Where $P$ is the pressure, $F$ is the force, $A$ is the area of force applied. 
\end{formula}
The standard unit for pressure $P$ is Pa:
\[\text{Pa}=\frac{\text N}{\text m^2}=\frac{\text{kg}}{\text{m}\cdot \text s^2}\]


Another commonly used unit of pressure is atm, which is the pressure of atmosphere at sea level. Here is how to convert pressure units:

\section{Boyle's Law}
\begin{formula}
    Boyle's Law is given by the formula,
    \[P_1V_1=P_2V_2\]
    Where $P_1$ is the initial pressure, $V_1$ is the initial volume, $P_2$ is the final pressure, $V_2$ is the final volume.
\end{formula}

\begin{custom-big}[Example 1]
Andi failed his math test, he throws a empty water bottle with pressure 1 atm to the ground and stamps on it. The volume is decreased to a third to its original. What is the pressure of the water bottle after Andi stamps on it?
\end{custom-big}
\begin{solution}
\begin{align*}
P_1V_1&=P_2V_2\\
1\text{ atm }V_1&=P_2\left(\frac 13V_1\right)\\
P_2&=3\text{ atm}
\end{align*}
\end{solution}

\section{Practices}
\begin{custom-big}[Example 2]
Lancy puts a balloon of 3.0 L in a sealed container, the initial pressure is 1.0 atm, and pumps the air out to 0.4 atm. Calculate the volume of the balloon. 
\end{custom-big}
\begin{solution}
\begin{align*}
P_1V_1&=P_2V_2\\
    1.0\text{ atm}\cdot 3.0\text{ L } &=0.4\text{ atm }V_2\\
    V_2&=7.5\text{ L}
\end{align*}
\end{solution}

\begin{custom-big}[Example 3]
A balloon has volume of 32.4 L, pressure of 1.00 atm, Janet presses the balloon with 500 kPa of pressure. What is the pressure and the volume of the balloon. \\
(1 atm = 101325 Pa)
\end{custom-big}
\begin{solution}
    
\begin{align*}
    P_2&=1\text{ atm}\cdot \frac{101325\text{ Pa}}{1\text{ atm}}+500000\text{ Pa}\\
    P_2&=\boxed{601000\text{ Pa}}\\
    P_1V_1&=P_2V_2\\
    101325\text{ Pa}\cdot 32.4\text{ L}&=P_2\cdot V_2\\
    V_2&=\frac{101325\text{ Pa}}{101325\text{ Pa}+500000\text{ Pa}}\cdot 32.4\text{ L}\\
    V_2&=\boxed{5.46\text{ L}}
\end{align*}
\end{solution}

\begin{custom-big}[Example 4]
It is known that the density of air is 1.293 kg per cubic metre, and Andi is 65.0 kg. Andi stands on a 1.00L air cube with pressure 1.00 atm. What is the pressure, volume, and density of the air cube after Andi stands on it? \\
(1 atm = 101325 Pa, g = 9.81 m/s$^2$)
\end{custom-big}

\begin{solution}

\begin{align*}
    P_{2}&=\frac{65\text{ kg }g \text{ m/s}^2}{0.01\text{ m}^2}+101325\text{ Pa}\\
    P_2&=\boxed{165000\text{ Pa}}\\
    P_1V_1&=P_2V_2\\
    101325\text{ Pa}(1\text{ L})&=165090\text{ Pa}\cdot V_2\\
    V_2&=\left(\frac{101325\text{ Pa}}{165090 \text{ Pa}}\right)1\text{ L}\\
    V_2&=\boxed{0.614\text{ L}}\\
    \rho_2&=\frac{1.293\text{ kg/m}^3\cdot \frac{1\text{ m}^3}{100\text{ L}}\cdot 1\text{ L}}{V_2\cdot \frac{1\text{ m}^3}{100\text{ L}}}\\
    \rho_2&=\boxed{2.11\text{ kg/m}^3}
\end{align*}
\end{solution}
