\begin{custom-big}[Problem 1]
A stunt vehicle leaves an incline with a speed of 35 m s at a height of 52 m above level ground. Air resistance is negligible.

\begin{center}
    \includegraphics[width=15cm]{Grade 11 Physics/Final Exam/kinematics.png}
\end{center}
\end{custom-big}

\pagebreak
\begin{solution} [Kinematics]Answer: \boxed{\text{Car successfully crosses the river}}\\

\begin{align*}
    v_{ix}&=35\cos{28^\circ}\\
    v_{iy}&=35\sin{28^\circ}\\
\end{align*}
\begin{center}
    \includegraphics[width=5cm]{Grade 11 Physics/Final Exam/vt.png}
\end{center}
The slope of the $v_y$ graph is the gravitational acceleration.\\
In the $y$-direction:
\begin{align*}
    35\sin{28^\circ}+\frac 12(-g)t^2&=-52\\
    \frac g2t^2-35\sin{28^\circ}t-52&=0\\
    t&=\frac{35\sin\left(28^\circ\right)+\sqrt{\left(35\sin\left(28^\circ\right)\right)^{2}+2g\left(52\right)}}{g}\\
    t&\approx 5.336
\end{align*}
Therefore the total distance traveled is $35\cos{28^\circ}(5.336)\approx 165$m $>150$m. The car will cross the river.
\end{solution}

\begin{custom-big}[Problem 2]
The two diagrams show a person sliding a block across a rough floor. Which method (a or b) shown in the diagram is easier for the person?
\begin{center}
    \includegraphics[width=15cm]{Grade 11 Physics/Final Exam/force.png}
\end{center}
\end{custom-big}

\begin{solution}[Force Question 1]
\begin{center}
    \includegraphics[width=5cm]{Grade 11 Physics/Final Exam/fa.png}
    \includegraphics[width=5cm]{Grade 11 Physics/Final Exam/fb.png}
\end{center}
\[
\begin{cases}
F_{N_a}=F_g+\sin{25^\circ}F_A\\
F_{N_b}+\sin{25^\circ}F_A=F_g
\end{cases}
\implies
\begin{cases}
F_{N_a}=F_g+\sin{25^\circ}F_A\\
F_{N_b}=F_g-\sin{25^\circ}F_A
\end{cases}
\]
\begin{align*}
    \because F_A&>0\\
    \therefore F_{N_a}&>F_{N_b}\\
    \therefore \mu F_{N_a}&>\mu F_{N_b}\\
    \therefore F_{f_a}&>F_{f_b}\\
    \therefore \text{b is}&\text{ easier}
\end{align*}
According to Newton's second law, $F=ma$. Because there is a greater force with constant mass, Object B will accelerate faster and eventually leaves the ground if the $y$ component of the applied force overcomes the gravity. 
\end{solution}

\begin{custom-big}[Problem 3]
A 4.0 kg block of wood sits on a table. A string is tied to the wood, running over a pulley and down to a hanging object. A hanging mass that causes the system to move is 1.8 kg. The coefficient of kinetic friction between the 4.0 kg object and the table is 0.30.
\end{custom-big}

\pagebreak
\begin{solution}[Force Question 2]
\begin{center}
    \includegraphics[width=5cm]{Grade 11 Physics/Final Exam/fbd.png}
\end{center}
\begin{align*}
    \sum F_x&=F_T-F_f\\
    \sum F_y&=0\\
    F_f&\propto F_N\\
\end{align*}

\[
\begin{cases}
\sum F_x=F_T-F_f\\
F_T=1.8g\\
F_f=\mu F_N\\
F_N=4g
\end{cases}
\implies
\sum F_x=1.8g-0.3(1.8g)=4a
\implies
a\approx 1.47 m/s^2
\]
\end{solution}
